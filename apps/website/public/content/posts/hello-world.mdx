### JavaScript

```javascript
function greet(name) {
  return `Hello, ${name}!`;
}

const message = greet('World');
console.log(message); // Outputs: Hello, World!
```

### TypeScript

```typescript
interface User {
  id: number;
  name: string;
  email?: string;
}

function getUserInfo(user: User): string {
  return `User ${user.name} (ID: ${user.id})`;
}

const user: User = {
  id: 1,
  name: 'John Doe'
};

console.log(getUserInfo(user));
```
